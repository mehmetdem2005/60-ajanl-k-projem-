### 📄 4. `packages/registry/src/server.ts`
Registry mantığını Fastify HTTP endpoint'lerine bağlayan ana sunucu dosyası.

```typescript
// packages/registry/src/server.ts
// Central Registry HTTP Server
// Exposes @planner/core RegistryService via REST API for distributed agent discovery.
// Handles registration, heartbeat, discovery, listing, and deregistration.

import Fastify from 'fastify';
import cors from '@fastify/cors';
import { RegistryService, AgentEndpoint } from '@planner/core';

// Singleton Registry Instance
const registry = new RegistryService();

// Initialize Fastify
const app = Fastify({ 
      logger: { 
            level: process.env.LOG_LEVEL || 'info',
                transport: { target: 'pino-pretty', options: { translateTime: 'HH:MM:ss Z' } }
      } 
});

// Register CORS
app.register(cors, { origin: true });

// Health Endpoint
app.get('/health', async (request, reply) => {
      return { 
            status: 'ok', 
                service: 'registry',
                    timestamp: Date.now(),
                        uptime_ms: process.uptime() * 1000
      };
});

// POST /api/v1/register
app.post<{ Body: AgentEndpoint }>('/api/v1/register', async (request, reply) => {
      try {
            const endpoint = request.body;
                registry.register(endpoint);
                    return { status: 'registered', agent_id: endpoint.agent_id, timestamp: Date.now() };
      } catch (error) {
            return reply.code(400).send({ error: 'Invalid registration payload', message: error instanceof Error ? error.message : 'Unknown' });
      }
});

// POST /api/v1/heartbeat/:agentId
app.post<{ Params: { agentId: string } }>('/api/v1/heartbeat/:agentId', async (request, reply) => {
      const { agentId } = request.params;
        registry.heartbeat(agentId);
          return { status: 'ok', agent_id: agentId, timestamp: Date.now() };
});

// GET /api/v1/discover/:agentId
app.get<{ Params: { agentId: string } }>('/api/v1/discover/:agentId', async (request, reply) => {
      const agent = registry.discover(request.params.agentId);
        if (!agent) {
                return reply.code(404).send({ error: 'Agent not found or offline', agent_id: request.params.agentId });
        }
          return agent;
});

// GET /api/v1/agents?capability=string
app.get<{ Querystring: { capability?: string } }>('/api/v1/agents', async (request, reply) => {
      const { capability } = request.query;
        const agents = registry.list(capability);
          return { 
                agents, 
                    count: agents.length, 
                        timestamp: Date.now() 
          };
});

// DELETE /api/v1/deregister/:agentId
app.delete<{ Params: { agentId: string } }>('/api/v1/deregister/:agentId', async (request, reply) => {
      registry.deregister(request.params.agentId);
        return { status: 'deregistered', agent_id: request.params.agentId, timestamp: Date.now() };
});

export default app;
```

# 🛠️ OPERASYONEL YÖNETİM KATMANI

### 📄 1. `Makefile`
Production ve development ortamlarında build, start, stop, log, test ve health-check işlemlerini standartlaştıran master yönetim dosyası.

⚠️ **Önemli:** `Makefile` dosyası girintilerde **TAB** karakteri kullanır. Kopyaladığınızda editörünüz boşluk yerine TAB kullandığından emin olun.

```makefile
# Makefile
# Planner Agent Ecosystem - Operational Management
# Usage: make <target> [VARIABLE=value]

.PHONY: help install build start stop down restart logs status health test clean migrate deploy prune

DOCKER := docker compose
NPM := npm
FILE := docker-compose.yml

help: ## 📖 Show all available commands
	@grep -E '^[a-zA-Z_-]+:.*?## .*$$' $(MAKEFILE_LIST) | sort | awk 'BEGIN {FS = ":.*?## "}; {printf "\033[36m%-15s\033[0m %s\n", $$1, $$2}'

install: ## 📦 Install npm dependencies (workspace)
	$(NPM) ci

build: ## 🏗️ Build TypeScript packages & Docker images
	$(NPM) run build
	$(DOCKER) -f $(FILE) build --no-cache

start: ## 🚀 Start all services (detached)
	$(DOCKER) -f $(FILE) up -d --remove-orphans

stop: ## 🛑 Stop all services gracefully
	$(DOCKER) -f $(FILE) stop

down: ## 🧹 Stop & remove containers, networks
	$(DOCKER) -f $(FILE) down

restart: ## 🔄 Restart all services
	$(DOCKER) -f $(FILE) restart

logs: ## 📜 Tail all logs (Ctrl+C to exit)
	$(DOCKER) -f $(FILE) logs -f --tail=200

logs-agent: ## 🔍 Tail specific agent logs (make logs-agent AGENT=planner)
	$(DOCKER) -f $(FILE) logs -f --tail=100 agent-$(AGENT)

status: ## 📊 Show container status
	$(DOCKER) -f $(FILE) ps

health: ## 💚 Verify registry & agent health
	@echo "🔍 Querying Registry Service..."
	@curl -sf http://localhost:3001/api/v1/agents | jq '.agents[] | {agent_id, status}' || echo "⚠️  Registry unreachable. Run 'make start' first."

test: ## 🧪 Run workspace unit tests
	$(NPM) run test

test-e2e: ## 🔄 Run end-to-end saga validation
	$(NPM) run test:e2e

clean: ## 🗑️ Remove containers, volumes & build artifacts
	$(DOCKER) -f $(FILE) down -v --remove-orphans
	rm -rf packages/*/dist
	rm -rf node_modules

migrate: ## 🔄 Run core migration & DB check
	@echo "⚙️  Running agent migration script..."
	$(NPM) run migrate
	@echo "✅ Migration complete."

deploy: ## 🌐 Production deploy (build → test → start)
	@echo "🚀 Starting production deployment sequence..."
	$(MAKE) test
	$(MAKE) build
	$(MAKE) start
	$(MAKE) health
	@echo "✅ Deployment successful."

prune: ## 🧽 Remove all unused Docker data
	docker system prune -af --volumes
```

### 📄 2. `scripts/verify-stack.sh`
Sistem ayağa kalktıktan sonra tüm ajanların registry'e kaydolduğunu, sağlık durumlarını ve ACP bağlantılarını doğrulayan otomatik validation scripti.

```bash
#!/usr/bin/env bash
# scripts/verify-stack.sh
# Automated Post-Deployment Verification Script
# Usage: chmod +x scripts/verify-stack.sh && ./scripts/verify-stack.sh

set -e

REGISTRY_URL="http://localhost:3001"
EXPECTED_AGENTS=("planner" "builder" "qa" "deployer" "monitoring" "incident" "cost" "compliance" "notification")

echo "🔍 Verifying Planner Agent Ecosystem Deployment..."
echo "=================================================="

# 1. Check Docker Status
if ! docker compose ps --format json 2>/dev/null | grep -q "running"; then
  echo "❌ Docker containers are not running. Run 'make start' first."
  exit 1
fi
echo "✅ Docker services are up."

# 2. Check Registry Health
if ! curl -sf "$REGISTRY_URL/health" > /dev/null; then
  echo "❌ Registry service is not healthy."
  exit 1
fi
echo "✅ Registry service is healthy."

# 3. Verify Agent Registration
echo "📡 Checking agent registrations..."
REGISTERED=$(curl -sf "$REGISTRY_URL/api/v1/agents" | jq -r '.agents[].agent_id')
MISSING=0

for agent in "${EXPECTED_AGENTS[@]}"; do
  if echo "$REGISTERED" | grep -q "${agent}.v1"; then
    echo "  ✅ ${agent}.v1 registered"
  else
    echo "  ❌ ${agent}.v1 missing"
    MISSING=$((MISSING + 1))
  fi
done

if [ $MISSING -gt 0 ]; then
  echo "⚠️  $MISSING agents failed to register. Check logs: make logs-agent AGENT=<name>"
  exit 1
fi

echo "=================================================="
echo "🎉 ALL SYSTEMS OPERATIONAL. Ecosystem is ready."
```

{{- range .Values.agents }}
apiVersion: apps/v1
kind: Deployment
metadata:
  name: {{ include "planner-ecosystem.name" $ }}-{{ .name }}
  labels:
    {{- include "planner-ecosystem.labels" $ | nindent 4 }}
    agent.id: {{ .id }}
spec:
  replicas: {{ .replicas }}
  selector:
    matchLabels:
      {{- include "planner-ecosystem.selectorLabels" $ | nindent 6 }}
      agent.id: {{ .id }}

    metadata:
      labels:
        {{- include "planner-ecosystem.labels" $ | nindent 8 }}
        agent.id: {{ .id }}
      annotations:
        checksum/config: {{ include (print $.Template.BasePath "/configmap.yaml") $ | sha256sum }}
    spec:
      securityContext:
        runAsNonRoot: true
        fsGroup: 1000
      containers:
        - name: {{ .name }}
          image: "{{ $.Values.global.image.repository }}/agent-{{ .name }}:{{ $.Values.global.image.tag }}"
          imagePullPolicy: {{ $.Values.global.image.pullPolicy }}
          ports:
            - containerPort: {{ .port }}
              name: http
              protocol: TCP
          env:
            - name: AGENT_ID
              value: {{ .id }}
            - name: PORT
              value: "{{ .port }}"
            - name: NODE_ENV
              value: {{ $.Values.global.env }}

              value: {{ $.Values.infra.nats.url | quote }}

            {{- toYaml . | nindent 12 }}
            {{- end }}
          resources:
            requests:
              {{- toYaml .resources.requests | nindent 16 }}

              {{- toYaml .resources.limits | nindent 16 }}

            failureThreshold: 2

        {{- toYaml . | nindent 8 }}

### 📄 6. `templates/networkpolicy.yaml`
Sıfır Güven (Zero-Trust) ağ izolasyonu. Varsayılan olarak tüm trafiği reddeder, yalnızca tanımlı portlara izin verir.

```yaml
# packages/infra/helm/planner-ecosystem/templates/networkpolicy.yaml
{{- if .Values.networkPolicy.enabled }}
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: {{ include "planner-ecosystem.name" . }}-default-deny
spec:
  podSelector: {}
  policyTypes:
    - Ingress
    - Egress
---
apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: {{ include "planner-ecosystem.name" . }}-allow-internal
spec:
  podSelector:
    matchExpressions:
      - { key: agent.id, operator: Exists }
  policyTypes:
    - Ingress
    - Egress
  ingress:
    - from:
        - podSelector:
            matchExpressions:
              - { key: agent.id, operator: Exists }
      ports:
        - { protocol: TCP, port: 3000 }
        - { protocol: TCP, port: 3001 }
  egress:
    - to:
        - podSelector:
            matchExpressions:
              - { key: agent.id, operator: Exists }
      ports:
        - { protocol: TCP, port: 3000 }
        - { protocol: TCP, port: 3001 }
    - to: []
      ports:
        - { protocol: TCP, port: 5432 }  # Postgres
        - { protocol: TCP, port: 6379 }  # Redis
        - { protocol: TCP, port: 4222 }  # NATS
        - { protocol: UDP, port: 53 }    # DNS
{{- end }}
```

1. **Helm Repo'ya Paketleyin:**
   ```bash
   helm package packages/infra/helm/planner-ecosystem/ --destination ./dist/charts
   ```

2. **Cluster'a Yükleyin:**
   ```bash
   # Namespace oluştur
   kubectl create namespace planner-system
   
   # Secret'ları önceden oluşturun (Production'da External Secrets Operator kullanın)
   kubectl create secret generic planner-secrets -n planner-system --from-literal=OPENAI_API_KEY=sk-...
   kubectl create secret generic builder-secrets -n planner-system --from-literal=GITHUB_TOKEN=ghp_...
   
   # Chart'ı kurun
   helm install planner-ecosystem ./dist/charts/planner-ecosystem-1.0.0.tgz \
     --namespace planner-system \
     -f values.yaml \
     --wait --timeout=10m
   ```

3. **Doğrulayın:**
   ```bash
   kubectl get pods -n planner-system
   kubectl get hpa -n planner-system
   curl -sf http://planner-ecosystem-registry.planner-system.svc.cluster.local:3001/api/v1/agents | jq
   ```
