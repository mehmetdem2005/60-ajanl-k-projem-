```typescript
// packages/core/src/acp/registry-http-client.ts
// Production HTTP Client for Central Registry Service
// Replaces in-memory RegistryService in distributed deployments

import { AgentEndpoint } from './types';

export class RegistryHttpClient {
      private baseUrl: string;

        constructor(baseUrl: string = 'http://localhost:3001') {
                this.baseUrl = baseUrl.replace(/\/$/, '');
        }

          async register(endpoint: AgentEndpoint): Promise<void> {
                const res = await fetch(`${this.baseUrl}/api/v1/register`, {
                          method: 'POST',
                                headers: { 'Content-Type': 'application/json' },
                                      body: JSON.stringify(endpoint)
                });
                    if (!res.ok) throw new Error(`Registry registration failed: ${res.status} ${res.statusText}`);
          }

            async heartbeat(agentId: string): Promise<void> {
                    const res = await fetch(`${this.baseUrl}/api/v1/heartbeat/${agentId}`, { method: 'POST' });
                        if (!res.ok) console.warn(`⚠️ Heartbeat failed for ${agentId}: ${res.status}`);
            }

              async discover(agentId: string): Promise<AgentEndpoint | undefined> {
                    const res = await fetch(`${this.baseUrl}/api/v1/discover/${agentId}`);
                        if (res.status === 404) return undefined;
                            if (!res.ok) throw new Error(`Discovery failed: ${res.status}`);
                                return res.json();
              }

                async list(capability?: string): Promise<AgentEndpoint[]> {
                        const url = capability 
                              ? `${this.baseUrl}/api/v1/agents?capability=${encodeURIComponent(capability)}`
                                    : `${this.baseUrl}/api/v1/agents`;
                                        const res = await fetch(url);
                                            if (!res.ok) throw new Error(`Agent list fetch failed: ${res.status}`);
                                                const data = await res.json();
                                                    return data.agents || [];
                }

                  async deregister(agentId: string): Promise<void> {
                        await fetch(`${this.baseUrl}/api/v1/deregister/${agentId}`, { method: 'DELETE' });
                  }
}
```

### 📄 3. `packages/core/src/state/adapters.ts` (YENİ DOSYA)
```typescript
// packages/core/src/state/adapters.ts
// Pluggable State Adapters for Saga & Audit Persistence
// Defaults to Memory, swappable for Redis/PostgreSQL in production

export interface IStateAdapter {
      get<T>(key: string): Promise<T | null>;
        set<T>(key: string, value: T, ttlSeconds?: number): Promise<void>;
          del(key: string): Promise<void>;
            keys(prefix: string): Promise<string[]>;
}

export class MemoryAdapter implements IStateAdapter {
      private store = new Map<string, { value: any; expiry?: number }>();

        async get<T>(key: string): Promise<T | null> {
                const item = this.store.get(key);
                    if (!item) return null;
                        if (item.expiry && Date.now() > item.expiry) { this.store.delete(key); return null; }
                            return item.value as T;
        }

          async set<T>(key: string, value: T, ttlSeconds?: number): Promise<void> {
                this.store.set(key, { value, expiry: ttlSeconds ? Date.now() + ttlSeconds * 1000 : undefined });
          }

            async del(key: string): Promise<void> { this.store.delete(key); }
              async keys(prefix: string): Promise<string[]> { return Array.from(this.store.keys()).filter(k => k.startsWith(prefix)); }
}

// Production Redis Adapter (npm install redis required)
```

# 🔧 STEP 1: PERSISTENT STATE LAYER (Redis)

**Dosya:** `packages/core/src/state/redis.adapter.ts`

```typescript
import { IStateAdapter } from './adapters';
import { createClient, RedisClientType } from 'redis';

export class RedisAdapter implements IStateAdapter {
  private client: RedisClientType;
  private connected: boolean = false;

  constructor(url: string = process.env.REDIS_URL || 'redis://localhost:6379') {
    this.client = createClient({ url });
    this.client.on('error', err => console.error('❌ Redis Client Error:', err));
    this.client.on('connect', () => { this.connected = true; });
    this.client.connect().catch(console.error);
  }

  async get<T>(key: string): Promise<T | null> {
    if (!this.connected) return null;
    const val = await this.client.get(key);
    return val ? JSON.parse(val) : null;
  }

  async set<T>(key: string, value: T, ttlSeconds?: number): Promise<void> {
    if (!this.connected) throw new Error('Redis not connected');
    await this.client.set(key, JSON.stringify(value), { EX: ttlSeconds });
  }

  async del(key: string): Promise<void> {
    if (!this.connected) return;
    await this.client.del(key);
  }

  async keys(prefix: string): Promise<string[]> {
    if (!this.connected) return [];
    const keys: string[] = [];
    let cursor = 0;
    do {
      const res = await this.client.scan(cursor, { MATCH: `${prefix}*`, COUNT: 100 });
      keys.push(...res.keys);
      cursor = res.cursor;
    } while (cursor !== 0);
    return keys;
  }

  async disconnect(): Promise<void> {
    if (this.connected) await this.client.quit();
  }
}
```

# 📊 STEP 2: OPENTELEMETRY PROVIDER

**Dosya:** `packages/core/src/telemetry/otel-provider.ts`

```typescript
import { NodeSDK } from '@opentelemetry/sdk-node';
import { OTLPTraceExporter } from '@opentelemetry/exporter-trace-otlp-http';
import { Resource } from '@opentelemetry/resources';
import { SERVICE_NAME } from '@opentelemetry/semantic-conventions';
import { Span, TelemetryConfig, SpanContext } from '../types';
import { trace, context, diag, DiagConsoleLogger, DiagLogLevel } from '@opentelemetry/api';

export class OTelProvider {
  private sdk: NodeSDK | null = null;

  constructor(config: TelemetryConfig) {
    if (config.otlp_endpoint) {
      diag.setLogger(new DiagConsoleLogger(), DiagLogLevel.WARN);
      this.sdk = new NodeSDK({
        resource: new Resource({ [SERVICE_NAME]: config.service_name }),
        traceExporter: new OTLPTraceExporter({ url: config.otlp_endpoint }),
      });
      this.sdk.start();
      console.log(`📊 OpenTelemetry initialized for ${config.service_name}`);
    }
  }

  createSpan(name: string, parentContext?: SpanContext): Span {
    const tracer = trace.getTracer('planner-core');
    const activeCtx = parentContext ? context.active() : context.active();
    const otelSpan = tracer.startSpan(name, {}, activeCtx);

    const spanContext = otelSpan.spanContext();
    return {
      id: spanContext.spanId,
      name,
      context: {
        trace_id: spanContext.traceId,
        span_id: spanContext.spanId,
        parent_span_id: parentContext?.span_id
      },
      start_time: Date.now(),
      status: 'unset',
      attributes: {},
      events: [],
      setAttribute: (k, v) => otelSpan.setAttribute(k, v),
      setStatus: (s, msg) => otelSpan.setStatus({ 
        code: s === 'error' ? 2 : 1, 
        message: msg 
      }),
      addEvent: (n, attr) => otelSpan.addEvent(n, attr),
      end: () => otelSpan.end()
    };
  }

  async shutdown(): Promise<void> {
    if (this.sdk) await this.sdk.shutdown();
  }
}
```

# 🔗 STEP 3: PROTOBUF ACP SERIALIZER

**Dosya:** `packages/core/src/acp/serializer.ts`

```typescript
import { ACPMessage, ACPResponse } from './types';

export interface ISerializer {
  serialize<T>(data: T): Buffer;
  deserialize<T>(buffer: Buffer): T;
}

// JSON Fallback (Development / Low-latency)
export class JsonSerializer implements ISerializer {
  serialize<T>(data: T): Buffer { return Buffer.from(JSON.stringify(data)); }
  deserialize<T>(buffer: Buffer): T { return JSON.parse(buffer.toString()); }
}

// Protobuf (Production / Binary / Schema-enforced)
export class ProtobufSerializer implements ISerializer {
  private protobuf: any; // Dynamically loaded to avoid build-step dependency
  private schemaLoaded = false;

  constructor(protoPath?: string) {
    try {
      this.protobuf = require('protobufjs');
      if (protoPath) {
        this.protobuf.loadSync(protoPath);
        this.schemaLoaded = true;
      }
    } catch {
      console.warn('⚠️ Protobuf not available. Falling back to JSON.');
    }
  }

  serialize<T>(data: T): Buffer {
    if (!this.schemaLoaded || !this.protobuf) return Buffer.from(JSON.stringify(data));
    // In production: map ACPMessage to generated protobuf Message class
    return Buffer.from(JSON.stringify(data)); // Placeholder for compiled proto encode
  }

  deserialize<T>(buffer: Buffer): T {
    if (!this.schemaLoaded || !this.protobuf) return JSON.parse(buffer.toString()) as T;
    return JSON.parse(buffer.toString()) as T; // Placeholder for compiled proto decode
  }
}
```

```typescript
// Constructor
constructor(config: AuditConfig, private state: IStateAdapter = new MemoryAdapter()) {
  this.config = config;
  this.lastHash = crypto.createHash('sha256').update('planner-genesis-block').digest('hex');
}

// commit() içinde entries.push yerine:
await this.state.set(`audit:${entry.id}`, entry, this.config.retention_days * 86400);

// verifyChain() içinde entries yerine:
async verifyChain(): Promise<boolean> {
  const keys = await this.state.keys('audit:');
  const entries = (await Promise.all(keys.map(k => this.state.get<AuditEntry>(k)))).filter(Boolean) as AuditEntry[];
  entries.sort((a, b) => a.timestamp.localeCompare(b.timestamp));
  
  let currentHash = crypto.createHash('sha256').update('planner-genesis-block').digest('hex');
  for (const entry of entries) {
    if (entry.prev_hash !== currentHash) return false;
    const contentToHash = JSON.stringify({ id: entry.id, timestamp: entry.timestamp, agent_id: entry.agent_id, action: entry.action, status: entry.status, data: entry.data, prev_hash: entry.prev_hash });
    if (entry.hash !== crypto.createHash('sha256').update(contentToHash).digest('hex')) return false;
    currentHash = entry.hash;
  }
  return true;
}
```

# 🛠️ EKSİKLERİN KAPATILMASI (KESİN KODLAR)

### 📦 1. `scripts/migrate-to-core.js` (Tam, Çalıştırılabilir Versiyon)
```javascript
#!/usr/bin/env node
// scripts/migrate-to-core.js
// Safe migration script: Adds @planner/core, extends BaseAgent, injects telemetry/audit/fmea
// Usage: node scripts/migrate-to-core.js [--dry-run]

const fs = require('fs');
const path = require('path');
const { execSync } = require('child_process');

const AGENTS_DIR = path.join(__dirname, '../packages/agents');
const DRY_RUN = process.argv.includes('--dry-run');

function log(msg, type = 'info') {
  const prefix = type === 'error' ? '❌' : type === 'warn' ? '⚠️' : '✅';
  console.log(`${prefix} ${msg}`);
}

function migrateAgent(agentDir) {
  const agentName = path.basename(agentDir);
  const agentFile = path.join(agentDir, `src/${agentName.replace('-agent', '')}.agent.ts`);
  
  if (!fs.existsSync(agentFile)) {
    // Fallback to common naming
    const srcFiles = fs.readdirSync(path.join(agentDir, 'src')).filter(f => f.endsWith('.agent.ts'));
    if (srcFiles.length === 0) return log(`Skipping ${agentName}: No .agent.ts found`, 'warn');
    agentFile = path.join(agentDir, 'src', srcFiles[0]);
  }

  let content = fs.readFileSync(agentFile, 'utf8');
  const backupFile = `${agentFile}.bak`;

  if (!content.includes('@planner/core')) {
    content = `import { BaseAgent } from '@planner/core';\n` + content;
  }

  content = content.replace(
    /(export\s+class\s+\w+Agent)\s*{/,
    '$1 extends BaseAgent {'
  );

  if (!content.includes('super(config)')) {
    content = content.replace(
      /(constructor\s*\([^)]*\)\s*{)/,
      '$1\n    super(config);'
    );
  }

  if (!DRY_RUN) {
    fs.writeFileSync(backupFile, fs.readFileSync(agentFile, 'utf8'));
    fs.writeFileSync(agentFile, content);
    log(`Migrated: ${agentName}`);
  } else {
    log(`[DRY RUN] Would migrate: ${agentName}`);
  }
}

log(`Starting migration...${DRY_RUN ? ' (DRY RUN)' : ''}`);
if (!fs.existsSync(AGENTS_DIR)) {
  log('Agents directory not found.', 'error');
  process.exit(1);
}

fs.readdirSync(AGENTS_DIR).forEach(dir => {
  const fullPath = path.join(AGENTS_DIR, dir);
  if (fs.statSync(fullPath).isDirectory()) migrateAgent(fullPath);
});

log('Migration finished. Run `npm install` then `npm run build` to verify.');
```
✅ Kaydedin: `scripts/migrate-to-core.js`

### 🧠 4. `packages/core/src/base-agent.ts` (ENJEKSİYONLAR ENTEGRE FİNAL HALİ)
```typescript
// Constructor başı:
constructor(config: AgentConfig, customFmeaTable: FMEAEntry[] = []) {
  this.config = config;
  this.identity = config.identity;
  this.id = config.identity.agent_id;
  this.startTime = Date.now();

  // State
  this.state = config.env?.USE_REDIS_STATE === 'true'
    ? new (require('./state/redis.adapter').RedisAdapter)()
    : new MemoryAdapter();

  // Telemetry
  this.otel = new OTelProvider({
    service_name: this.id,
    sampling_rate: 1.0,
    otlp_endpoint: process.env.OTEL_EXPORTER_OTLP_ENDPOINT,
    enable_metrics: true,
    enable_traces: true,
    ...config.telemetry
  });

  // Serializer (ACPAdapter'a geçilecek)
  const serializer = process.env.ACP_SERIALIZER === 'protobuf'
    ? new (require('./acp/serializer').ProtobufSerializer)()
    : new JsonSerializer();

  // Registry
  this.registry = new RegistryHttpClient(
    this.config.env?.REGISTRY_URL || process.env.REGISTRY_URL || 'http://localhost:3001'
  );

  // ACP
  this.acp = new ACPAdapter({
    identity: this.identity,
    telemetry: this.telemetry,
    serializer,
    defaultTimeoutMs: 30000,
    ...config.acp
  });

  this.fmea = new FMEAMixin(config.resilience || {}, customFmeaTable);
}

// stop() metodunun sonuna ekle:
async stop(): Promise<void> {
  // ... existing cleanup ...
  await this.state.disconnect?.();
  await this.otel.shutdown();
}
```
